public class PlanTravailBlueprint {

    public PlanTravailBlueprint() {
        this.blueprintActif = null;
        this.noeudSelectionne = null;
        
        // Initialisation de la vue : centré avec un zoom normal
        this.cameraX = 0f;

    // Gère le déplacement dans l'espace (le "Pan" de la caméra)
    public void deplacerCamera(float deplacementX, float deplacementY) {
        this.cameraX += deplacementX;
        this.cameraY += deplacementY;
        // La logique future dira : "Demande à l'interface de rafraîchir l'écran"
    }

    // Gère le rapprochement ou l'éloignement (Pinch-to-zoom natif)
    public void ajusterZoom(float facteurZoom) {
        this.niveauZoom *= facteurZoom;
        
        // Sécurité pour éviter de zoomer trop près ou trop loin
        if (this.niveauZoom < 0.2f) this.niveauZoom = 0.2f;
        if (this.niveauZoom > 3.0f) this.niveauZoom = 3.0f;
